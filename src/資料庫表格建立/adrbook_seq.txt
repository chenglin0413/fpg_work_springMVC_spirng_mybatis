CREATE SEQUENCE ADRBOOK_SEQ
MINVALUE  200000001
MAXVALUE  209999991
INCREMENT BY 1
NOCACHE NOCYCLE;

查詢目前表格中有使用中sequence
 SELECT * FROM ADRBOOK_SEQ;

刪除sequence:
 DROP SEQUENCE ADRBOOK_SEQ;

●取用一組新的值:
 SELECT ADRBOOK_SEQ.NEXTVAL FROM DUAL;

得知目前使用到幾號 :
 SELECT ADRBOOK_SEQ.CURRVAL FROM DUAL;
